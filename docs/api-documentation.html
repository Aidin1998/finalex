// Calculate portfolio value
function calculatePortfolioValue(balances, prices) {
  return balances.reduce((total, balance) => {
    const price = prices[balance.currency] || 0;
    return total + (parseFloat(balance.available) * price);
  }, 0);
}

function calculateMovingAverage(prices, period) {
  return prices.slice(-period).reduce((sum, price) => sum + price, 0) / period;
}

Code Examples

Ready-to-use code examples in multiple programming languages.

JavaScript/Node.js

const WebSocket = require('ws');
const axios = require('axios');

class FinalexAPI {
  constructor(apiKey, secretKey) {
    this.apiKey = apiKey;
    this.secretKey = secretKey;
    this.baseURL = 'https://api.finalex.io';
    this.wsURL = 'wss://ws.finalex.io';
  }

  async authenticateWithJWT(email, password) {
    const response = await axios.post(`${this.baseURL}/api/v1/auth/login`, {
      email, password
    });
    this.token = response.data.access_token;
    return this.token;
  }

  connectWebSocket() {
    this.ws = new WebSocket(`${this.wsURL}?token=${this.token}`);
    
    this.ws.on('open', () => {
      console.log('Connected to Finalex WebSocket');
    });
    
    this.ws.on('message', (data) => {
      const message = JSON.parse(data);
      this.handleMessage(message);
    });
  }

  subscribe(channel, symbol = null) {
    const params = { channel };
    if (symbol) params.symbol = symbol;
    
    this.ws.send(JSON.stringify({
      method: 'subscribe',
      params
    }));
  }

  async placeOrder(symbol, side, type, quantity, price = null) {
    const orderData = { symbol, side, type, quantity };
    if (price) orderData.price = price;

    const response = await axios.post(
      `${this.baseURL}/api/v1/trading/orders`,
      orderData,
      { headers: { 'Authorization': `Bearer ${this.token}` } }
    );
    
    return response.data;
  }
}

// Usage
const api = new FinalexAPI('your_api_key', 'your_secret_key');
await api.authenticateWithJWT('user@example.com', 'password');
api.connectWebSocket();
api.subscribe('ticker', 'BTC/USDT');

Python

import asyncio
import websockets
import json
import requests
import hmac
import hashlib
import time

class FinalexAPI:
    def __init__(self, api_key=None, secret_key=None):
        self.api_key = api_key
        self.secret_key = secret_key
        self.base_url = 'https://api.finalex.io'
        self.ws_url = 'wss://ws.finalex.io'
        self.token = None

    def authenticate_jwt(self, email, password):
        response = requests.post(f'{self.base_url}/api/v1/auth/login', {
            'email': email,
            'password': password
        })
        self.token = response.json()['access_token']
        return self.token

    async def connect_websocket(self):
        uri = f"{self.ws_url}?token={self.token}"
        self.websocket = await websockets.connect(uri)
        
        # Start listening for messages
        asyncio.create_task(self.listen_messages())

    async def listen_messages(self):
        async for message in self.websocket:
            data = json.loads(message)
            await self.handle_message(data)

    async def subscribe(self, channel, symbol=None):
        params = {'channel': channel}
        if symbol:
            params['symbol'] = symbol
            
        message = {
            'method': 'subscribe',
            'params': params
        }
        
        await self.websocket.send(json.dumps(message))

    def place_order(self, symbol, side, order_type, quantity, price=None):
        order_data = {
            'symbol': symbol,
            'side': side,
            'type': order_type,
            'quantity': quantity
        }
        if price:
            order_data['price'] = price

        headers = {'Authorization': f'Bearer {self.token}'}
        response = requests.post(
            f'{self.base_url}/api/v1/trading/orders',
            json=order_data,
            headers=headers
        )
        
        return response.json()

    async def handle_message(self, message):
        print(f"Received: {message}")

# Usage
async def main():
    api = FinalexAPI()
    api.authenticate_jwt('user@example.com', 'password')
    await api.connect_websocket()
    await api.subscribe('ticker', 'BTC/USDT')
    
    # Keep connection alive
    await asyncio.sleep(3600)

asyncio.run(main())
